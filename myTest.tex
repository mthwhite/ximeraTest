\begin{question}
	What is the worst kind of cat?
	\begin{prompt}
		\begin{multipleChoice}
			\choice{tabby}
			\choice[correct]{puppy}
			\choice{dog}
			\choice{kitten}
			\choice{main coon}
		\end{multipleChoice}
	\end{prompt}
	\begin{hint}
		It is not a cat or a type of cat.
	\end{hint}
	\begin{hint}
		It is a puppy!
	\end{hint}
\end{question}

	
	\begin{question}
	Factor the following completely.\\
	$ x^2 + 2x + 1 $ 
	\begin{hint}
		$ x^2 + 2x + 1 = (x + 1)(x + 1) = (x + 1)^2 $
		\end{hint}
		\begin{prompt}
	\begin{multipleChoice}
		\choice{$x^2 + 2x + 1$}
		\choice[correct]{$\left(x + 1\right)^2$}
		\choice{wrong}
	\end{multipleChoice}
	\end{prompt}
	\end{question}

	\begin{question}
		What is one half of $x^2$?\\
		$	\answer{x^2/2}$
	\end{question}
