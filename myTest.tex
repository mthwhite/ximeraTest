  		Factor the following completely.\\
  		$ x^2 + 2x + 1 $ 
  		\begin{hint}
  			$ x^2 + 2x + 1 = (x + 1)(x + 1) = (x + 1)^2 $
  		\end{hint}
  		\begin{prompt}
  			\begin{multipleChoice}
  				\choice{$x^2 + 2x + 1$}
  				\choice[correct]{$\left(x + 1\right)^2$}
  				\choice{$(x - 1)(x + 1)$}
  			\end{multipleChoice}
  		\end{prompt}
  \end{question}
